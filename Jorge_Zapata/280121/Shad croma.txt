void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    // Normalized pixel coordinates (from 0 to 1)
    vec2 uv = fragCoord/iResolution.xy;

    // Time varying pixel color
    vec4 col_1 = texture(iChannel0, uv + vec2(0.01, 0.02));
    vec4 col_2 = texture(iChannel0, uv + vec2(-0.01, 0.02));
    vec4 col_3 = texture(iChannel0, uv + vec2(0.01, -0.02));
    
    vec4 fincol = vec4 (col_1.r,col_2.g, col_3.b, 1.0);

    // Output to screen
    fragColor = fincol;
}
